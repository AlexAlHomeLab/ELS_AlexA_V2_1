void sync_feed_start(float feed_per_rev) {
if (spindle_get_speed() == 0) return;
uint32_t freq = spindle_get_frequency();
uint32_t steps_per_rev = calculate_steps(feed_per_rev);
timer_set_frequency(freq * steps_per_rev);
}
